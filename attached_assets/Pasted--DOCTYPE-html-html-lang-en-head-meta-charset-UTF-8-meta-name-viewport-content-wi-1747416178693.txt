  <script>
    document.addEventListener('DOMContentLoaded', function () {
      const calendarEl = document.getElementById('calendar');
      const workoutView = document.getElementById('workout-view');
      const workoutTitle = document.getElementById('workout-title');
      const workoutDetails = document.getElementById('workout-details');
      const workoutFrame = document.getElementById('workout-frame');

      const calendar = new FullCalendar.Calendar(calendarEl, {
        initialView: 'dayGridMonth',
        headerToolbar: {
          left: 'prev,next today',
          center: 'title',
          right: ''
        },
        events: [
          {
            title: 'Day 1 – Full Body Start',
            start: '2025-05-20',
            extendedProps: {
              details: 'Seated Chest Press, Bicep Curls, Dumbbell Squats. Focus on form.',
              link: 'https://your-replit-app-url/day1.html'
            }
          },
          {
            title: 'Day 2 – Core & Cardio',
            start: '2025-05-22',
            extendedProps: {
              details: 'Marching on the spot, resistance band rows, 4-4 breathing.',
              link: 'https://your-replit-app-url/day2.html'
            }
          },
          {
            title: 'Day 3 – Rest & Recovery',
            start: '2025-05-24',
            extendedProps: {
              details: 'Mobility stretches and 4-4 breathing. Optional walk.',
              link: 'https://your-replit-app-url/day3.html'
            }
          }
        ],
        eventClick: function (info) {
          workoutTitle.textContent = info.event.title;
          workoutDetails.textContent = info.event.extendedProps.details;
          workoutFrame.src = info.event.extendedProps.link;
          workoutView.style.display = 'block';
          calendarEl.style.display = 'none';
        }
      });

      calendar.render();

      window.closeWorkout = function () {
        workoutView.style.display = 'none';
        calendarEl.style.display = 'block';
        workoutFrame.src = '';
      };
    });
  </script>
